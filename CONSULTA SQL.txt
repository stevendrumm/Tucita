SELECT DISTINCT PRO.NOMBRE AS NOMBRE, 
	   CITA.CODIGO AS CODPRO,
	   CITA.CENTROPROD AS CENTRO
	   FROM fac_p_profesional AS PRO, 
	   fac_p_rx_profcprod AS CENPRO, 
	   fac_m_citas AS CITA 
	   WHERE CENPRO.CODIGO = '1110' AND		
	   CENPRO.PROFESIONAL=PRO.CODIGO AND
	   CITA.CODIGO=PRO.CODIGO AND
	   CITA.CENTROPROD IS NULL
	   AND DATEPART(DD,CITA.FECHA) = DATEPART(DD,'2016-07-05')
	   AND DATEPART(MM,CITA.FECHA) = DATEPART(MM,'2016-07-05')
	   AND DATEPART(YYYY, CITA.FECHA) = DATEPART(YYYY,'2016-07-05')
	   ORDER BY CITA.CODIGO
